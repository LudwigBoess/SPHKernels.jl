SPHKernels.Cubic — Type
struct Cubic <: SPHKernel
    n_neighbours::Int64
    norm_1D::Float64
    norm_2D::Float64
    norm_3D::Float64
    function Cubic(n_neighbours::Int64=64)
        new(n_neighbours, 8.0/π, 8.0/π)
    end
end

Datatype for cubic sph spline.
